// Leer los pedidos de la Mesa 1 desde Firebase const pedidosRef = ref(database, 'pedidos/mesa1'); // Ruta para los pedidos de la Mesa 1 const pedidosContainer = document.getElementById('pedidos'); // Obtener los datos de la base de datos de Firebase get(pedidosRef).then((snapshot) => { if (snapshot.exists()) { // Si hay pedidos, los mostramos const pedidoData = snapshot.val(); const pedidoElement = document.createElement('div'); pedidoElement.textContent = `Plato: ${pedidoData.plato}, Fecha: ${pedidoData.fecha}`; pedidosContainer.appendChild(pedidoElement); } else { // Si no hay pedidos pedidosContainer.textContent = "No hay pedidos."; } }).catch((error) => { console.error("Error al leer los pedidos: " + error); }); // Leer los mensajes del chat de la Mesa 1 desde Firebase const chatRef = ref(database, 'chat/mesa1'); // Ruta para el chat de la Mesa 1 const chatMessagesContainer = document.getElementById('chatMessages'); // Obtener los mensajes del chat get(chatRef).then((snapshot) => { if (snapshot.exists()) { const chatData = snapshot.val(); const messageElement = document.createElement('div'); messageElement.textContent = chatData.mensaje; chatMessagesContainer.appendChild(messageElement); } else { chatMessagesContainer.textContent = "No hay mensajes en el chat."; } }).catch((error) => { console.error("Error al leer los mensajes del chat: " + error); });
