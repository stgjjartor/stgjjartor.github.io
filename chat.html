// Función para enviar el mensaje al chat document.getElementById('enviarMensajeBtn').addEventListener('click', function () { const mensaje = document.getElementById('mensajeCliente').value; // Verificamos que haya un mensaje antes de enviarlo if (mensaje.trim() !== "") { const chatRef = ref(database, 'chat/mesa1'); // Ruta para el chat de la Mesa 1 const nuevoMensaje = { mensaje: mensaje, fecha: new Date().toLocaleString() }; // Guardar el mensaje en Firebase set(chatRef, nuevoMensaje) .then(() => { alert("¡Mensaje enviado con éxito!"); document.getElementById('mensajeCliente').value = ""; // Limpiar el campo }) .catch((error) => { alert("Error al enviar el mensaje: " + error); }); } });
